Quick Start

# Install SeedML
pip install -e .
export ANTHROPIC_API_KEY='your-api-key'

# Create a simple app
cat > app.seed << EOF
app TodoApp {
  entity Task {
    title: string!
    done: bool = false
  }
  screen Tasks {
    list: [title, done]
    actions: [create, toggle-done]
  }
}
EOF

# Generate the application
seedml generate app.seed
